GIT

git config --global user.name "Name":- Set your name in git globally
git config --global user.email "email@gmail.com":- Set your email in git globally
git config user.name:- Give the user name
git config user.email:- Give the user email
git status:- Give the status of current directory
git init:- Initialize the directory as a git repository
ls:- List all the files and folders of current directory
pwd:- Give the path of present working directory
cd:- Change directory
git add --a or git add .:- Track all the files or staged all the files
git add filename:- Track or staged only the given file


git commit -m "Initial Commit":- Commit the state
git log:- Give details of all commits
git commit -a -m "Initial Commit":- Staged all the files and commit the state
rm filename:- Only delete the file
git rm filename:- Delete the file and also staged it
git mv filename file_rename:- Rename the file and staged it
git clone Given_link folder:- Clone the directory in a given folder from a given link
git log -p:- Show all the details of commit
git log -p -n:- Give all the details of last n commits
git log --pretty=oneline:- Give details of all commits in one line
git log --pretty=full:- Give full details of commit
git log --since=2.days:- Give details of commits since 2 days
git log --stat:- Give all stats about all commits
git commit --amend:- You can merge your last commit with the previous one
git remote:- Give the name of remote repository
git remote -v:- Give the name of pull and push remote repository
git remote add remote_name Given_link:- Add remote in your git bash
git restore filename:- Remove all the changes from the given file
git restore --staged filename:- Unstaged the given file
git checkout -- filename:- Match the file to the previous commit
git checkout -f:- Discard all changes and match with previous commit
ssh-keygen -t rsa -b 4096 -C "email@gmail.com":- Generate SSH key
eval "$(ssh-agent -s)":- Tell about your SSH agent id
ssh-add ~/.ssh/id_rsa:- Add your SSH private id
tail ~/.ssh/id_rsa.pub:- Write your SSH public id in your git bash
git push -u origin master:- Push your cod in github
